import { serve } from "https://deno.land/std@0.168.0/http/server.ts"; import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", }; serve(async (req) => { // Handle CORS preflight if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }); } try { const { to, subject, html, from } = await req.json(); // Get Resend API key from company_settings const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); const { data: settings } = await supabase .from("company_settings") .select("resend_api_key") .limit(1) .single(); const resendKey = settings?.resend_api_key; if (!resendKey) { return new Response( JSON.stringify({ error: "No Resend API key found in company_settings" }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } // Send via Resend const resendResp = await fetch("https://api.resend.com/emails", { method: "POST", headers: { "Authorization": `Bearer ${resendKey}`, "Content-Type": "application/json", }, body: JSON.stringify({ from: from || "FreightFlowHQ ", to: Array.isArray(to) ? to : [to], subject, html, }), }); const result = await resendResp.json(); if (!resendResp.ok) { return new Response( JSON.stringify({ error: result.message || "Resend API error", details: result }), { status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } return new Response( JSON.stringify({ success: true, id: result.id }), { status: 200, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } catch (err) { return new Response( JSON.stringify({ error: err.message }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } } ); } });
